doc_mover()
	{
	clear
	mkdir ~/"$2"
	cd "$1" || exit
	tar -cvzf "OG.$1.$(date +%F).tar.gz" *
	clear
	rsync -azPv  "OG.$1.$(date +%F).tar.gz" ~/"$2"
	cd ~/"$2" || exit
	tar -xvzf "OG.$1.$(date +%F).tar.gz"
	clear
	diff ~/"$1" ~/"$2"
	cd || exit
	}
```

trackDNS()
{
clear
IP=$(dig a "$1" +short)
NSIP=$(dig ns "$1" +short)
MX=$(dig mx "$1" +short)
TXT=$(dig txt "$1" +short)
RED='\033[0;31m'
NC='\033[0m' # No Color
clear
echo -e "${RED}IP INFORMATION for $1 ${NC}"
curl ipinfo.io/"$IP"
echo -e "${RED}Auth A, MX and TXT records  @$NSIP for $1 ${NC}"
dig a "$1" @"$NSIP"  +short
dig mx "$1" @"$NSIP" +short
dig txt "$1" @"$NSIP"  +short
echo -e "${RED}REGISTRY for $1 ${NC}"
whois "$1" | grep 'Name Server\|Expiry\|Domain Status'
curl -IL  "$1" | head -1
ping -c4 "$1"
traceroute "$1"
}
```
